// This script handles the Weekly Timesheet form ONLY.
// The Office Supplies form lives in a separate Apps Script project
// (also bound to this same Google Sheet), with its own deployment URL.
//
// Changes from v15:
//   - readProjects() now exposes a "label" field per project (pass-through
//     from the new Sheet column "Label", written by MondaySync v5).
//   - Type still drives the form section; label drives the prefix tag.
//   - If label is empty, the form falls back to the type for the prefix.

const CONSOLIDATED_SHEET = 'Consolidated';
const PROJECTS_SHEET = 'Projects';
const PLANNING_SHEET = 'NextWeekPlanning';
const ADHOC_TASKS_SHEET = 'AdHocExamples';
const FAQ_SHEET = 'FAQ';

// Optional Slack incoming webhook URL. Empty = disabled.
const SLACK_WEBHOOK_URL = '';

// =====================================================================
// GET - returns project list / users / ad-hoc tasks / FAQ as JSON
// =====================================================================
function doGet(e) {
  try {
    const action = (e.parameter && e.parameter.action) || '';
    if (action === 'projects') {
      const projects = readProjects(false);
      return ContentService
        .createTextOutput(JSON.stringify({ ok: true, projects }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    if (action === 'allProjects') {
      const projects = readProjects(true);
      return ContentService
        .createTextOutput(JSON.stringify({ ok: true, projects }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    if (action === 'users') {
      const users = readActiveUsers();
      return ContentService
        .createTextOutput(JSON.stringify({ ok: true, users }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    if (action === 'adhocTasks') {
      const tasks = readAdHocTasks();
      return ContentService
        .createTextOutput(JSON.stringify({ ok: true, tasks }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    if (action === 'faq') {
      const faq = readFAQ();
      return ContentService
        .createTextOutput(JSON.stringify({ ok: true, faq }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: 'Unknown action' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: String(err) }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function readProjects(includeInactive) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(PROJECTS_SHEET);
  if (!sheet) return [];
  const last = sheet.getLastRow();
  if (last < 2) return [];

  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const nameIdx = headers.indexOf('Project');
  const typeIdx = headers.indexOf('Type');
  const activeIdx = headers.indexOf('Active');
  const labelIdx = headers.indexOf('Label');  // NEW (added in v16, written by MondaySync v5)
  if (nameIdx === -1 || typeIdx === -1 || activeIdx === -1) return [];

  const data = sheet.getRange(2, 1, last - 1, sheet.getLastColumn()).getValues();
  return data
    .filter(r => {
      if (!r[nameIdx]) return false;
      if (includeInactive) return true;
      return String(r[activeIdx]).trim().toLowerCase() === 'yes';
    })
    .map(r => {
      const typeStr = String(r[typeIdx]).trim().toLowerCase();
      let type = 'sprint';
      if (typeStr === 'régie' || typeStr === 'regie') type = 'regie';
      else if (typeStr === 'ad-hoc' || typeStr === 'adhoc' || typeStr === 'ad hoc') type = 'adhoc';
      const label = labelIdx === -1 ? '' : String(r[labelIdx] || '').trim();
      return {
        name: String(r[nameIdx]).trim(),
        type: type,
        label: label,                                 // pass-through from Sheet, may be ''
        active: String(r[activeIdx]).trim().toLowerCase() === 'yes'
      };
    });
}

// Read active users from the TeamDirectory tab (synced by MondaySync.gs).
// Returns just the email field as a sorted array of strings.
function readActiveUsers() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName('TeamDirectory');
  if (!sheet) return [];
  const last = sheet.getLastRow();
  if (last < 2) return [];

  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const emailIdx = headers.indexOf('Email');
  if (emailIdx === -1) return [];

  const data = sheet.getRange(2, 1, last - 1, sheet.getLastColumn()).getValues();
  const emails = data
    .map(r => String(r[emailIdx] || '').trim().toLowerCase())
    .filter(e => e && e.indexOf('@') !== -1);

  return Array.from(new Set(emails)).sort();
}

// Read ad-hoc task examples from AdHocExamples tab.
// Schema: ONE column. Row 1 = 'Example' (header), rows 2+ = examples.
// Returns: flat array of strings, e.g. ['Setting up gateway...', 'Onboarding...']
function readAdHocTasks() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(ADHOC_TASKS_SHEET);
  if (!sheet) return [];
  const last = sheet.getLastRow();
  if (last < 2) return [];

  // Start at row 2 to skip the header row
  const data = sheet.getRange(2, 1, last - 1, 1).getValues();
  return data
    .map(row => String(row[0] || '').trim())
    .filter(s => s.length > 0);
}

// Read FAQ from FAQ tab.
// Schema: column A = Question, column B = Answer.
// Row 1 = headers ('Question' / 'Answer'), always skipped.
// Returns: flat array of {question, answer} objects.
function readFAQ() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(FAQ_SHEET);
  if (!sheet) return [];
  const last = sheet.getLastRow();
  if (last < 2) return [];

  // Start at row 2 to skip the header row
  const data = sheet.getRange(2, 1, last - 1, 2).getValues();
  const items = [];

  data.forEach(row => {
    const q = String(row[0] || '').trim();
    const a = String(row[1] || '').trim();
    if (!q || !a) return;  // skip empty rows or rows missing either side
    items.push({ question: q, answer: a });
  });

  return items;
}

// =====================================================================
// POST - receives a timesheet submission
// =====================================================================
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(CONSOLIDATED_SHEET);

    if (!sheet) {
      sheet = ss.insertSheet(CONSOLIDATED_SHEET);
      sheet.appendRow([
        'SubmittedAt', 'Email', 'WeekStart', 'SubmissionNumber', 'Category', 'Project',
        'TaskOrActivity', 'Days', 'Hours', 'WFHNextWeek', 'LeaveNextWeek',
        'ManualTotal', 'AutoFill', 'Comments'
      ]);
      sheet.getRange(1, 1, 1, 14)
           .setFontWeight('bold').setBackground('#0F2A3D').setFontColor('white');
      sheet.setFrozenRows(1);
    }

    const incomingEmail = String(data.email || '').trim().toLowerCase();
    const incomingWeek = normalizeDateStr(data.weekStart);

    // Count + remove existing rows for this user-week
    let priorSubmissions = 0;
    const lastRow = sheet.getLastRow();
    if (lastRow > 1) {
      const existing = sheet.getRange(2, 2, lastRow - 1, 3).getValues();
      const seenNumbers = new Set();
      for (let i = existing.length - 1; i >= 0; i--) {
        const rEmail = String(existing[i][0] || '').trim().toLowerCase();
        const rWeek = normalizeDateStr(existing[i][1]);
        if (rEmail === incomingEmail && rWeek === incomingWeek) {
          seenNumbers.add(existing[i][2]);
          sheet.deleteRow(i + 2);
        }
      }
      priorSubmissions = seenNumbers.size;
    }
    const submissionNumber = priorSubmissions + 1;

    // Append new rows
    const rows = (data.entries || []).map(en => [
      data.submittedAt,
      data.email,
      data.weekStart,
      submissionNumber,
      en.category,
      en.project || '',
      en.taskOrActivity,
      en.days,
      en.days * 8, // Hours = Days * 8
      data.nextWfh || '',
      data.nextLeave || '',
      data.manualTotal,
      data.autoFill,
      data.comments || ''
    ]);
    if (rows.length > 0) {
      sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, 14).setValues(rows);
    }

    // Mirror next-week planning to its own dedicated tab
    if (data.nextWfh || data.nextLeave) {
      let planning = ss.getSheetByName(PLANNING_SHEET);
      if (!planning) {
        planning = ss.insertSheet(PLANNING_SHEET);
        planning.appendRow([
          'SubmittedAt', 'Email', 'WeekStartReferenced', 'NextWeekStart',
          'WFHNextWeek', 'LeaveNextWeek'
        ]);
        planning.getRange(1, 1, 1, 6)
                .setFontWeight('bold').setBackground('#3FA6B8').setFontColor('white');
        planning.setFrozenRows(1);
      }

      let nextWeekStart = '';
      try {
        const d = new Date(incomingWeek + 'T00:00:00');
        d.setDate(d.getDate() + 7);
        nextWeekStart = d.toISOString().slice(0, 10);
      } catch (e) {}

      const pLastRow = planning.getLastRow();
      if (pLastRow > 1) {
        const pData = planning.getRange(2, 2, pLastRow - 1, 2).getValues();
        for (let i = pData.length - 1; i >= 0; i--) {
          const rEmail = String(pData[i][0] || '').trim().toLowerCase();
          const rWeek = normalizeDateStr(pData[i][1]);
          if (rEmail === incomingEmail && rWeek === incomingWeek) {
            planning.deleteRow(i + 2);
          }
        }
      }

      planning.appendRow([
        data.submittedAt,
        data.email,
        data.weekStart,
        nextWeekStart,
        data.nextWfh || '',
        data.nextLeave || ''
      ]);
    }

    sendRecapEmail(data);
    if (SLACK_WEBHOOK_URL) sendSlackPing(data);

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true, rows: rows.length, submissionNumber }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: String(err) }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// =====================================================================
// SLACK
// =====================================================================
function sendSlackPing(data) {
  const isOver = data.manualTotal > 5.0001;
  const prefix = isOver ? '[OVERTIME]' : (data.autoFill > 0 ? '[AUTO]' : '[OK]');
  const text = `${prefix} *Timesheet submitted* by ${data.email} · week of ${data.weekStart} · ${formatNum(data.manualTotal)}d manual + ${formatNum(data.autoFill)}d auto`;
  try {
    UrlFetchApp.fetch(SLACK_WEBHOOK_URL, {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify({ text }),
      muteHttpExceptions: true
    });
  } catch (err) {
    Logger.log('Slack ping failed: ' + err);
  }
}

// HELPERS
// =====================================================================
function formatNum(n) {
  if (typeof n !== 'number') n = parseFloat(n) || 0;
  if (n === 0) return '0';
  return parseFloat(n.toFixed(4)).toString();
}

function escapeHtml(s) {
  return String(s || '').replace(/[&<>"']/g, ch => ({
    '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'
  }[ch]));
}

function extractFirstName(email) {
  if (!email) return '';
  const local = String(email).split('@')[0];
  if (!local) return '';
  const first = local.split(/[._-]+/)[0];
  if (!first || first.length < 2) return '';
  return first.charAt(0).toUpperCase() + first.slice(1).toLowerCase();
}

function normalizeDateStr(v) {
  if (!v) return '';
  if (v instanceof Date) {
    return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  return String(v).slice(0, 10);
}
